    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const temperatureInput = document.getElementById('temperature');
            const convertBtn = document.getElementById('convert-btn');
            const resultElement = document.getElementById('result');
            const formulaElement = document.getElementById('formula');
            const errorElement = document.getElementById('temperature-error');
            
            convertBtn.addEventListener('click', convertTemperature);
            
            function convertTemperature() {
                // Reset error state
                temperatureInput.classList.remove('input-error');
                errorElement.style.display = 'none';
                
                // Get input value
                const inputValue = parseFloat(temperatureInput.value);
                
                // Validate input
                if (isNaN(inputValue)) {
                    temperatureInput.classList.add('input-error');
                    errorElement.style.display = 'block';
                    resultElement.textContent = '-';
                    formulaElement.textContent = '';
                    return;
                }
                
                // Get conversion types
                const fromType = document.querySelector('input[name="from"]:checked').value;
                const toType = document.querySelector('input[name="to"]:checked').value;
                
                // If converting to the same unit
                if (fromType === toType) {
                    resultElement.textContent = `${inputValue} ${getUnitSymbol(toType)}`;
                    formulaElement.textContent = 'No conversion needed - same units';
                    return;
                }
                
                // Perform conversion
                let result;
                let formula;
                
                // Conversion formulas
                if (fromType === 'celsius') {
                    if (toType === 'fahrenheit') {
                        result = (inputValue * 9/5) + 32;
                        formula = `Formula: (${inputValue} °C × 9/5) + 32 = ${result.toFixed(2)} °F`;
                    } else if (toType === 'kelvin') {
                        result = inputValue + 273.15;
                        formula = `Formula: ${inputValue} °C + 273.15 = ${result.toFixed(2)} K`;
                    }
                } else if (fromType === 'fahrenheit') {
                    if (toType === 'celsius') {
                        result = (inputValue - 32) * 5/9;
                        formula = `Formula: (${inputValue} °F - 32) × 5/9 = ${result.toFixed(2)} °C`;
                    } else if (toType === 'kelvin') {
                        result = (inputValue - 32) * 5/9 + 273.15;
                        formula = `Formula: (${inputValue} °F - 32) × 5/9 + 273.15 = ${result.toFixed(2)} K`;
                    }
                } else if (fromType === 'kelvin') {
                    if (toType === 'celsius') {
                        result = inputValue - 273.15;
                        formula = `Formula: ${inputValue} K - 273.15 = ${result.toFixed(2)} °C`;
                    } else if (toType === 'fahrenheit') {
                        result = (inputValue - 273.15) * 9/5 + 32;
                        formula = `Formula: (${inputValue} K - 273.15) × 9/5 + 32 = ${result.toFixed(2)} °F`;
                    }
                }
                
                // Display result
                resultElement.textContent = `${result.toFixed(2)} ${getUnitSymbol(toType)}`;
                formulaElement.textContent = formula;
            }
            
            function getUnitSymbol(unit) {
                switch(unit) {
                    case 'celsius': return '°C';
                    case 'fahrenheit': return '°F';
                    case 'kelvin': return 'K';
                    default: return '';
                }
            }
        });
